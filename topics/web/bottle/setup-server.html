Tool to install and configure a Tableau Server cluster
 -a,--authType <type>                       Authentication type, LOCAL (default) or ACTIVEDIRECTORY.
 -b,--branch <name>                         Branch installer, e.g. server-dev. Use the latest unless build is specified.
    --build <number>                        Build number, e.g. 15.0829.2025. Used with branch option.
    --buildType <type>                      Specify a build type, NIGHTLY, CONTINUOUS, or REMOTE. Default is all build
                                            types if a build number is specified, or official builds only (nightly or
                                            continuous) if looking for the latest build of a branch.
 -c,--clean                                 Clean, delete the Tableau content in Program Files and ProgramData
 -cc,--cleanClusteredNodes                  Clean all nodes that are in an existing cluster with any of the install
                                            nodes.False to only uninstall and/or clean the install nodes.
    --config <filename>                     Path to config topology file
 -D <option=value>                          Set a server option (and can set as many as you want), e.g.
                                            -Dbackgrounder.log.level=debug -Dfilestore.log.level=debug
    --dataDir <dir>                         Complete path to customer data directory
 -dbpass,--remoteDBPass <password>          Password to enable remote repository access for the 'tableau' user.
    --devLicense                            Activate Tableau Server with dev internal license key
                                            (TS4H-TEST-GSERVER-PERM)
    --disableAccountCreation                disable account creation for TSM
    --disableStatistics                     disable statistics collection for SetupServer
    --enableRemoteDBConnForTSM              Configure remote db connections for TSM
    --enableZookeeperHAForTSM               Configure zookeeper HA for TSM
 -f,--fasterUpgrade                         Utilize the '/fastuninstall' flag of the installer, allowing the installer
                                            to perform uninstall without taking a backup. Works for 10.00 and higher
                                            releases.
 -fu,--forceUninstall                       Force Uninstalls the primary and worker node/s. Supports clean and non
                                            default location flags during uninstallation. Pretends the uninstall was
                                            successful even if its fails
 -h,--help                                  Print help
 -i,--installerDir <dir>                    Private build installer directory
 -id,--installDir <dir>                     Complete path to installation directory (where server binaries will be
                                            copied). Overrides the default path for install.
    --initialPassword <value>               The initial password, override the default of workgroupuser password.
    --initialUsername <value>               The initial username, override the default of workgroupuser
    --installSamples <arg>                  Server installer will add samples
 -l,--logPath <log>                         Specify the custom path of log to store in format path/name. The log path
                                            string should include both the directory and name of the log. e.g.,
                                            'C:\Program Files\Tableau\log1.txt'. Note this log is the product installer
                                            log, but not related with the SetupServer App log during installation.
 -lf,--licenseFile <file>                   Path to file to license Tableau Server with. Currently only supports asr
                                            files.
 -lic,--licenses <key>                      License keys to activate Tableau Server with
 -lt,--licenseType <type>                   Use an internal license key of the provided type: 'named_user', 'core',
                                            'registration', or 'none'
 -n,--nodes <primary worker1 worker2 ...>   Node names, space-delimited list, first one is the primary
 -p,--password <value>                      RunAs password, override the default tsi\workgroupuser password.
    --privilegedUser <value>                Set a privileged user for TSM
 -q,--quiet                                 Don't output log messages
    --rc <number>                           Release candidate number, e.g. 6 (for rc6). Used with version option.
    --readonlyDBPass <password>             Password to enable remote repository access for the 'readonly' user.
    --reg <filename>                        Path to registration file
 -s,--secrets <filename>                    Path to secrets file
    --serverAuthGroup <value>               Set a server authorized group for TSM
    --serverInstaller <filename>            Server installer filename, e.g. myServer.exe. Used with installerDir option
                                            to specify a non-default name.
    --skipWorkers                           Skip running anything on workers. This is useful for non-domain-joined
                                            environments (like AWS). If enabled, it is the responsibility of the caller
                                            to ensure the workers are properly installed at the right time.
 -ssl,--sslCertPath <dir>                   Complete path to ssl certificates directory (where ssl certs for this
                                            cluster are found). Can be a local or remote path on machine. Certs will be
                                            copied to Tableau Server folder under the installation Directory
 -tsm,--useTSMInstaller                     Uses the  TSM installers for Server and Worker installation
    --tsmAdminPassword <value>              TSM Admin password, override the default tsi\tabauto password
    --tsmAdminUsername <value>              TSM admin username, override the default of tsi\tabauto
 -u,--username <value>                      RunAs username, override the default of tsi\workgroupuser
 -ud,--uninstallDir <dir>                   Complete path to installation directory (where server uninstall binary is
                                            found) for uninstall. Overrides the default path for uninstall.
    --unprivilegedUser <value>              Set an unprivileged user for TSM
 -uo,--uninstallonly                        Uninstalls the primary and worker node/s. Supports clean and non default
                                            location flags during uninstallation
 -v,--version <number>                      Version installer, e.g. 9.0.2. Use the latest unless rc is specified.
    --workerInstaller <filename>            Worker installer filename, e.g. myWorker.exe. Used with installerDir option
                                            to specify a non-default name.
    --x86                                   Specify 32-bit installer.
 -y,--yml <file>                            Path to configuration yml file (e.g. tabsvc.yml)
